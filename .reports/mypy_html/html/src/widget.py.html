import re
from datetime import datetime
from typing import Union

from masks import get_mask_account, get_mask_card_number


def mask_account_card(input_string: str) -> str:

    match = re.match(r"(.*)\s+(\d+)", input_string)
    if not match:
        raise ValueError("Неверный формат ввода")

    name_part, number_part = match.groups()

    try:
        number = int(number_part)

        # Определение типа и применение соответствующей маски
        if any(keyword in name_part for keyword in ["Visa", "Maestro", "MasterCard"]):
            masked_number = get_mask_card_number(number)
        else:
            masked_number = get_mask_account(number)

        return f"{name_part} {masked_number}"
    except Exception as e:
        raise ValueError(f"Ошибка обработки данных: {e}")


def get_date(iso_date: str) -> str:

    dt = datetime.fromisoformat(iso_date)
    return dt.strftime("%d.%m.%Y")
